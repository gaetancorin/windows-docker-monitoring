-------------- AIDE DOCKER COMPOSE POSTGRE ---------------
construire le docker compose:
docker-compose -p monitoring_streaming up -d

forcer a tout rebuild:
docker-compose -p monitoring_streaming up --build -d

supprimer tout les containers + les volumes
docker-compose -p monitoring_streaming down -v

redemarrer le docker compose:
docker-compose -p monitoring_streaming start

arreter le docker compose:
docker-compose -p monitoring_streaming stop

supprimer le docker compose et docker associés:
docker-compose -p monitoring_streaming down

recréer le docker compose:
docker-compose -p monitoring_streaming down
docker-compose -p monitoring_streaming up -d
